import androidx.appcompat.app.AppCompatActivity;

import android.os.Bundle;
import android.view.View;

public class MainActivity extends AppCompatActivity {

            @Override
            public void onClick(View v) {
                TextView tv = findViewById(R.id.tv);
                tv.setText(Talk(1, 2));

        });
    }

    String Talk(int n1, int n2){
        return "Talk:" + String.valueOf(n1 + n2);
    }
   
}

function main() {
    console.log("start hooking ");
    Java.perform(function() {
        console.log("Inside java perform function");
        //定位类
        var MainActivity = Java.use("org.example.demo01.MainActivity");
        console.log("Java.Use.Successfully!"); //定位类成功！

        // 使用 `implementation` 关键词实现实例化对应
        MainActivity.Talk.implementation = function (x, y) {
            console.log("MainActivity.Talk:", x, y);            
            // 调用原有的方法
            // return this.Talk(x, y);            
            // 替换参数
            return this.Talk(10, 20);

        }
    });
}

function main() {
    console.log("start hooking ");
    Java.perform(function() {
        console.log("Inside java perform function");
        //定位类
        var MainActivity = Java.use("org.example.demo01.MainActivity");
        console.log("Java.Use.Successfully!"); //定位类成功！

        // 静态函数调用
        MainActivity.staticHello()

        // 实例函数调用
        // `Java.choose` 是在内存中查找对应的类，前提是要保证类已经加载到内存中；
        // 运行脚本前，需要先运行应用；
        Java.choose("org.example.demo01.MainActivity", {
            onMatch: function(instance) {
                console.log("instance:", instance);
                instance.hello();
            },
            onComplete: function() {
                console.log("onComplete");
            }
        });
    }
}

function CallHello() {
    Java.perform(function() {
        Java.choose("org.example.demo01.MainActivity", {
            onMatch: function(instance) {
                instance.hello();
            },
            onComplete: function() {
            }
        });
    }
}

rpc.exports = {
    // 导出名（key）不可以有大写字母或者下划线!!!
    callhello: CallHello,
}

import frida
import sys
import time

def on_message(message, data):
    if message['type'] == 'send':
        print("[*] {0}".format(message['payload']))
    else:
        print(message)


device = frida.get_usb_device()
process = device.attach("com.example.demo01")

with open("hook.js") as f:
    jsCode = f.read()
script = process.create_script(jsCode)

script.on('message', on_message)
script.load()

while True:
    script.exports.callhello()
    time.sleep(1)

//hook.js
Java.perform(function () {
    console.log("Inside java perform function");
    // 通过 hook TextView 控件
    Java.use("android.widget.TextView").setText.overload('java.lang.CharSequence').implementation = function (x) {
        // 将参数发送给 Python 脚本
        var strSend = x.toString();
        send(strSend);

        // 接收 Python 脚本返回的结果
        var strRecv;
        recv(function (received_json_objection) {
            strRecv = received_json_objection.mData;
            console.log("strRecv ", strRecv)
        }).wait()

        // 更新UI
        var javaStr = Java.use('java.lang.String').$new(strRecv);
        var result = this.setText(javaStr)
        console.log("javaStr result", javaStr, result)

        return result
    }
})
